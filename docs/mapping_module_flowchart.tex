%% ── Row 0: sensor inputs ─────────────────────────────────────────────────────
\node[sens] (lidar) at (-3.4, 0) {LiDAR Scan\\[2pt] 360 pts \textbf{·} 10 Hz};
\node[sens] (odom)  at ( 3.4, 0) {Wheel Odometry\\[2pt] 10 Hz};

%% ── Row 1: parallel processing ───────────────────────────────────────────────
\node[feat] (feat) at (-3.4, -2.8)
  {\textbf{Feature Extraction}\\[3pt]
   TLS incremental grow + adjacent merge\\
   Walls $(\rho,\alpha)$ \textbf{·} Corners $(x,y)$\\
   CRLB / Jacobian covariance};

\node[ekf]  (pred) at ( 3.4, -2.8)
  {\textbf{EKF Prediction}\\[3pt]
   \texttt{predict\_with\_relative\_motion}\\
   midpoint integration\\
   motion-scaled process noise $\mathbf{Q}$};

%% merge coordinate (feat+pred → assoc)
\coordinate (mfp) at (0, -4.8);

%% ── Row 2: data association ───────────────────────────────────────────────────
\node[feat] (assoc) at (0, -6.2)
  {\textbf{Data Association}\\[3pt]
   Euclidean pre-filter (6 m radius)\\
   Mahalanobis gate \;$D^2_M < 5.99$\\
   wall $\alpha/\rho$ tolerance \textbf{·} endpoint gap check};

%% ── Row 3: decision ───────────────────────────────────────────────────────────
\node[dec]  (dmatch) at (0, -8.5) {Feature\\matched?};

%% ── Row 4: matched / unmatched branches ──────────────────────────────────────
\node[ekf]  (update) at (-3.6, -11.0)
  {\textbf{EKF Update}\\[3pt]
   Joseph form covariance update\\
   wall $\rho\!\geq\!0$ normalisation\\
   full Hessian sync to FeatureMap};

\node[feat] (addlm) at ( 3.6, -11.0)
  {\textbf{New Landmark}\\[3pt]
   \texttt{add\_landmark()}\\
   2-scan observation threshold\\
   initialise in EKF state + FeatureMap};

%% merge coordinate (branches → prune)
\coordinate (mbrch) at (0, -13.0);

%% ── Row 5: prune ─────────────────────────────────────────────────────────────
\node[ekf]  (prune) at (0, -13.9)
  {\textbf{Prune Landmarks}\\[3pt]
   remove if unseen $>\!25$ scans\\
   sync deletion to FeatureMap};

%% ── Row 6: feature map update ────────────────────────────────────────────────
\node[map]  (fmap)  at (0, -15.9)
  {\textbf{FeatureMap Update}\\[3pt]
   wall arc-length extents $(t_{\min},\,t_{\max})$ — \textbf{persistent}\\
   corner positions read from EKF state};

%% ── Row 7: submap boundary decision ──────────────────────────────────────────
\node[dec]  (d50)   at (0, -18.0) {Submap\\boundary?\\(50 scans)};

%% ─── dashed separator ────────────────────────────────────────────────────────
\draw[gray!40, dashed, line width=0.8pt]
  (-8.0,-19.1) -- (8.0,-19.1);
\node[gray!70, font=\scriptsize\itshape, anchor=east]
  at (7.9,-18.75) {Submap pipeline \textit{(every 50 scans)}};

%% ═════════════════════════════════════════════════════════════════════════════
%%  SUBMAP PIPELINE
%% ═════════════════════════════════════════════════════════════════════════════

%% ── Row 8: generate point cloud ──────────────────────────────────────────────
\node[map]  (genpc) at (0, -20.2)
  {\textbf{Generate Point Cloud}\\[3pt]
   \texttt{generate\_point\_cloud(0.05 m)}\\
   walls + corners \textbf{·} voxel-downsample\\
   extents always valid — no reset needed};

%% ── Row 9: SVD alignment ─────────────────────────────────────────────────────
\node[map]  (svd)   at (0, -22.4)
  {\textbf{SVD Feature Alignment}\\[3pt]
   shared wall IDs $\;\to\;$ correspondence pairs\\
   one-shot Procrustes SVD solve\\
   condition-number degeneracy gate};

%% ── Row 10: SVD success decision ─────────────────────────────────────────────
\node[dec]  (dsvd)  at (0, -24.6) {SVD\\success?};

%% ── Row 11: EKF correction (Yes branch) ──────────────────────────────────────
\node[ekf]  (corr)  at (-3.6, -26.9)
  {\textbf{EKF Correction}\\[3pt]
   $\mathbf{R}_{\mathrm{SVD}}$-weighted pose observation\\
   correct EKF map-frame drift\\
   \texttt{LandmarkEKFSLAM.update()}};

%% ── Row 12: global map output ────────────────────────────────────────────────
\node[out]  (gmap)  at (0, -29.4)
  {\textbf{Global Map}\\[3pt]
   concatenate + voxel-downsample (0.05 m)\\
   GPU-accelerated (Open3D tensor API)\\
   published on \texttt{/{robot}/global\_map} @ 1 Hz};

%% assoc → decision

\draw[arr] (dmatch.west) -- ++(-0.8,0)
  node[above, font=\scriptsize]{Yes}
  |- (update.west);
\draw[arr] (dmatch.east) -- ++(0.8,0)
  node[above, font=\scriptsize]{No}
  |- (addlm.east);

\draw[arr] (update.south) |- (mbrch);
\draw[arr] (addlm.south)  |- (mbrch);

\draw[arr] (genpc) -- (svd);
\draw[arr] (svd)   -- (dsvd);

%% dsvd → corr  (Yes)
\draw[arr] (dsvd.west) -- ++(-0.8,0)
  node[above, font=\scriptsize]{Yes}
  |- (corr.west);

%% dsvd → gmap  (No — use EKF pose as-is)
\draw[arr] (dsvd.east) -- ++(2.6,0)
  node[right, font=\scriptsize]{No\\(EKF pose)}
  |- (gmap.east);

%% corr → gmap
\draw[arr] (corr.south) -- ++(0,-0.6) -| (gmap.west);
